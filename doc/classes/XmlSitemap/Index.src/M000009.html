    def render
      output = [] ; map_id = 1
      output << '<?xml version="1.0" encoding="UTF-8"?>'
      output << '<sitemapindex xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">'
      @maps.each do |m|
        output << '<sitemap>'
        output << "<loc>#{m[:loc]}</loc>"
        output << "<lastmod>#{m[:lastmod]}</lastmod>"
        output << '</sitemap>'
      end
      output << '</sitemapindex>'
      return output.join("\n")
    end
